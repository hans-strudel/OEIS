%o A053087 a053087 n = head [j | j <- [0..], (2 ^ j) `kara` n /= Nothing] where
%o A053087    kara a b = if null ks then Nothing else Just $ head ks
%o A053087               where ks = [c | c <- [1..a], a <= c * b, a > c * (b - 1)]
