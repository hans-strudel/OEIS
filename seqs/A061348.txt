%p A061348 A061348 := proc(n) local t1, v, a; a := n*(n+1)/2; v := floor((n+1)/2); if n mod 3 = 1 then t1 := n*(n+1)/6+2/3; else t1 := n*(n+1)/6; fi; (1/6)*(2^a + 2*2^t1+3*2^(a/2+v/2)); end; # from Burnside's Lemma
