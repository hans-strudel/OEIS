%o A050367            m = #digits(n, k) - 1;

%o A050367            for(i=0, m, w[k^i] = polcoeff(A, i));
%o A050367 \\ build the answer
%o A050367            v = dirmul(v, w)
%o A050367       );
%o A050367 v[n]
%o A050367   )
%o A050367 };
