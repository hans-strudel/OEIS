%o A030699 Q(n) = {
%o A030699   my(q = vector(n));
%o A030699   q[1] = [1,0,0,0];
%o A030699   for (i = 2, n,
%o A030699        my(m = floor((sqrt(8*i+1) - 1)/2));
%o A030699        q[i] = vector((1 + (m>>2)) << 2);
%o A030699        q[i][1] = 1;
%o A030699        for (j = 2, m, q[i][j] = q[i-j][j] + q[i-j][j-1]));
%o A030699   return(q);
%o A030699 };
%o A030699 N = 59;
%o A030699 apply(vecmax, Q(N))  \\ _Gheorghe Coserea_, Nov 04 2015
