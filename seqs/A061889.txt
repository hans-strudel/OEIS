%N A061889 a(n) = number of solutions to x_1 + x_2 + ... + x_n = x_{n+1} where each x_i is a factorial.
%e A061889 a[6]=3 since 5!+5!+5!+5!+5!+5!=6!, 3!+3!+3!+2!+2!+2!=4! and 1!+1!+1!+1!+1!+1!=3!
%t A061889 f[n_, k_, m_] := (* # of partitions of n into k factorials <= m! *) Which[n==k, 1, m<=1||n<k|| n>k m!, 0, True, f[n, k, m]=f[n, k, m-1]+f[n-m!, k-1, m]]; a[n_] := Sum[f[r!, n, r-1], {r, 2, n}];
