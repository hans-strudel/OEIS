%N A037383 Numbers n such that every base 3 digit of n is a base 7 digit of n.

%o A037383 a037383 n = a037383_list !! (n-1)
%o A037383 a037383_list = filter f [1..] where
%o A037383    f x = null $ nub (ds 3 x) \\ nub (ds 7 x)
